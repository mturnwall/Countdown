 23 var Countdown = /** @lends Countdown.prototype */{
 24     /**
 25      *   find the difference in days, hours, mins, seconds between two dates.
 26      *   @param {Array} eventDate the date counting down to, [yyyy, m, d]
 27      *   the month is zero based so January is 0, December is 11.
 28      */
 29     getTimeDifference: function(eventDate) {
 30         var difference,
 31                 endDate,
 32                 oneDay,
 33                 oneHour,
 34                 oneMinute,
 35                 oneSecond,
 36                 startDate,
 37                 values;
 38 
 39         values = [];
 40         oneDay = 1000 * 60 * 60 * 24;
 41         oneHour = 1000 * 60 * 60;
 42         oneMinute = 1000 * 60;
 43         oneSecond = 1000;
 44         if (eventDate.length < 4) {
 45             endDate = new Date(eventDate[0], eventDate[1], eventDate[2]);
 46         } else {
 47             endDate = new Date(eventDate[0], eventDate[1], eventDate[2], eventDate[3], eventDate[4]);
 48         }        
 49         startDate = new Date();
 50         
 51         /** subtract the event date with today's date minus the timezone difference */
 52         difference = endDate.getTime() - (startDate.getTime() - this.findTimezone());
 53 
 54         // Thanks to Richard Rudzinski for reminding me about modulus!
 55         // days
 56         if (difference >= oneDay) {
 57             this.timeValues['days'] = Math.floor(difference / oneDay);
 58         } else {
 59             this.timeValues['days'] = 0;
 60         }
 61         // hours
 62         this.timeValues['hours'] = Math.floor((difference % oneDay) / oneHour);
 63         // minutes
 64         this.timeValues['mins'] = Math.floor(((difference % oneDay) % oneHour) / oneMinute);
 65         // seconds
 66         this.timeValues['seconds'] = Math.ceil((((difference % oneDay) % oneHour) % oneMinute) / oneSecond);
 67     },
 68     /**
 69      *  handle just countdowning hours and minutes
 70      */
 71     startTimeCounter: function (eventDate) {
 72         var extraMins;
 73         this.getTimeDifference(eventDate);
 74         if (this.timeValues['hours']) {
 75             extraMins = this.timeValues['hours'] * 60;  // take any extra hours and add them to the total minutes
 76             this.timeValues['mins'] = this.timeValues['mins'] + extraMins;
 77         }
 78     },
 79 
 80     /**
 81      *   Find the current timezone the user's computer is in and then move it back to PDT
 82      *   @return {Integer} the total difference between current timezone and PDT in milliseconds
 83      */
 84     findTimezone: function() {
 85         var now, currentZone;
 86 
 87         now = new Date();
 88         currentZone = now.getTimezoneOffset();
 89         if (currentZone == this.pacificOffset) {
 90             return 0;
 91         }
 92 
 93         if (currentZone < 0) {
 94             currentZone = (currentZone * -1) + this.pacificOffset;
 95         } else if (currentZone < this.pacificOffset) {
 96             currentZone = this.pacificOffset - currentZone;
 97         } else {
 98             currentZone = this.pacificOffset + currentZone;
 99         }
100 
101         return currentZone * 1000 * 60;
102     },
103 
104     /**
105      *   take current time value and deduct a second
106      */
107     countTime: function() {
108         // temp variables so we have a copy to manipulate
109         var days = this.timeValues['days'],
110             hours = this.timeValues['hours'],
111             mins = this.timeValues['mins'],
112             secs = this.timeValues['seconds'],
113             time;
114         if (secs > 0) {
115             secs--;
116         } else if (secs == 0 && mins > 0) {
117             secs = 59;
118             mins--;
119         } else if (secs == 0 && mins == 0 && hours > 0) {
120             if (mins == 0 && hours > 0) {
121                 secs = 59;
122                 mins = 59;
123             }
124             if (hours > 1) {
125                 hours--;
126             } else if (hours == 1 && days > 0) {
127                 hours = 0;  // it is possile to have 1 day 00 hours 59 mins
128                 days--;
129             } else {
130                 hours = 0;
131             }
132         }
133         time = {
134             days: (days >= 0) ? days : 0,
135             hours: (hours >= 0) ? hours : 0,
136             mins: (mins >= 0) ? mins : 0,
137             seconds: (secs >= 0) ? secs : 0
138         };
139         /** write the new time values back to the main time array for next time */
140         for (var i in time) {
141             this.timeValues[i] = time[i];
142         };
143         this.writeTime(time);
144     },
145 
146     /**
147      *   write the new time values into the page
148      */
149     writeTime: function(values) {
150         var obj = this;
151         this.$_timeContainers.each(function(i) {
152             var timeType = this.parentNode.className.match(/days|hours|mins|seconds/)[0];
153             // add leading zero if needed
154             if (values[timeType] >= 0) {
155                 (values[timeType].toString().length > 1) ? this.innerHTML = values[timeType] : this.innerHTML = '0' + values[timeType];
156             }
157         });
158         // only continue countdown if all values aren't zero
159         if (values['days'] != 0 || values['hours'] != 0 || values['mins'] != 0 || values['seconds'] != 0) {
160             this.timer = window.setTimeout(function() {
161                 obj.countTime();
162             }, 1000);
163         } else {
164             return window.location.reload();
165         }
166     },
167     /** 
168      *   initializes the object
169      *   @param opts options for the object to initialize with
170      *   @param {Array} opts.eventDate the date counting down to, [yyyy, m, d, hh, mm, ss]
171      *       the month is zero based so January is 0, December is 11.
172      *   @param {Boolean} opts.levelTimer true if the timer is used for a level timer instead of
173              an upcoming event timer
174      *   @param {Boolean} opts.dst whether or not daylight savings is currently in affect
175      *   @param {String} opts.status a message to be displayed
176      */
177     init: function(opts) {
178         this.$_timeContainers = $('.timerCont .timeValue span'); // HTML elements that hold the time values
179         if (opts.eventStatus) {
180             $('.timerCont').html(opts.eventStatus);
181             return true;
182         }
183         this.timeValues = [];
184         this.timer;     // setTimeout timer
185         this.dst = (typeof opts.dst == 'undefined') ? false : opts.dst;
186         this.pacificOffset = (this.dst) ? 420 : 480;    // find out if we have PST or PDT
187         if (!opts.levelTimer) {
188             this.getTimeDifference(opts.eventDate);  // find the difference between event date and today's date
189         } else {
190             this.startTimeCounter(opts.eventDate);
191         }
192         this.writeTime(this.timeValues);    // write the difference to the page and start countdown
193     }
194 };
